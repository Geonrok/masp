## 전략 코드

```python
import pandas as pd

def check_entry_signal(df):
    """
    Multi_TF_Short 진입 신호 확인
    df: 최근 60일 이상의 OHLCV + 외국인 데이터
    """
    # 단기 외국인 비중 MA
    wght_ma5 = df['wght'].rolling(5).mean().iloc[-1]
    wght_ma20 = df['wght'].rolling(20).mean().iloc[-1]

    # 중기 외국인 비중 MA
    wght_ma10 = df['wght'].rolling(10).mean().iloc[-1]
    wght_ma40 = df['wght'].rolling(40).mean().iloc[-1]

    # 가격 MA
    price_ma50 = df['close'].rolling(50).mean().iloc[-1]
    current_price = df['close'].iloc[-1]

    # 조건 확인
    short_up = wght_ma5 > wght_ma20
    mid_up = wght_ma10 > wght_ma40
    price_up = current_price > price_ma50

    return short_up and mid_up and price_up
```
